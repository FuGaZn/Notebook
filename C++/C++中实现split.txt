vector<string> split(string& s, const string& c){
	vector<string> vec;
	std::string::size_type pos1, pos2;
	pos2 = s.find(c);
	pos1 = 0;
	while (string::npos != pos2){
		vec.push_back(s.substr(pos1, pos2 - pos1));
		pos1 = pos2 + c.size();
		pos2 = s.find(c, pos1);
	}
	if (pos1 != s.length()){
		vec.push_back(s.substr(pos1));
	}
	return vec;
}
